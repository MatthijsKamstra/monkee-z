// Setup the component
    var appJson = new MonkeeChain('#app', {
        data: {
            json: {
                name: '',
                size: {
                    minified: "",
                    original: "",
                    uglifyjs: "",
                }
            }
        },
        template: function (props) {
            return `
            File ${props.json.name}:
            
    
                
  • Original file size: ${props.json.size.original}
    • 
                
  • UglifyJs file size: ${props.json.size.uglifyjs}
    • 
                
  • Extra minified file size: ${props.json.size.minified}
    • 
            

            `;
        }
    });

    // Fetch API data
    fetch('json/monkee_load.json').
        then(function (response) {
            return response.json();
        }).then(function (data) {
            //console.log(data);
            appJson.data.json = data;
            appJson.render();
        });
